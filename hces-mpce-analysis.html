# Function to compute quintiles within each sector
compute_fractiles <- function(data) {
  data <- data %>% arrange(MPCE) %>%
    mutate(pop = Weights.x * Household.size.F,
           cumul_pop = cumsum(pop),
           total_pop = sum(pop, na.rm = TRUE),
           pop_percentage = (cumul_pop / total_pop) * 100,
           fractile = case_when(
             pop_percentage <= 20 ~ "0-20",
             pop_percentage > 20 & pop_percentage <= 40 ~ "20-40",
             pop_percentage > 40 & pop_percentage <= 60 ~ "40-60",
             pop_percentage > 60 & pop_percentage <= 80 ~ "60-80",
             pop_percentage > 80 ~ "80-100"
           ))
  return(data)
}
